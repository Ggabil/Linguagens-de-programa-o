        int pot1 = 0;
        
        int pot2 = 0;
        
        int res = 0;
        
        void setup()
        {
          pinMode(12, OUTPUT);
          pinMode(A0, INPUT);
          Serial.begin(9600);
          pinMode(A1, INPUT);
        
          pot1 = 0;
          pot2 += 0;
          res = 0;
          digitalWrite(12, LOW);
        }
        
        void loop()
        {
          pot1 += analogRead(A0);
          Serial.println("pot1--------------------------");
          Serial.println(pot1);
          Serial.println("pot2--------------------------");
          Serial.println(pot2);
          pot2 += analogRead(A1);
          if (pot1 > pot2) {
            res += (pot1 + pot2);
            Serial.println("res-----------------------------");
            Serial.println(res);
            digitalWrite(12, HIGH);
          } else {
            digitalWrite(12, LOW);
          }
          delay(1); // Wait for 1 millisecond(s)
        }
